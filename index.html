Google Sheet Sync

One-time setup (~5 min). After this, every saved deal appends a row to your spreadsheet automatically.

Step 1

Open your Google Sheet → Extensions → Apps Script

Step 2 · Paste this script, then Save (💾)
function doPost(e) { try { var data = JSON.parse(e.postData.contents); var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; sheet.appendRow([ data.customerName, // Customer Name data.business, // Business data.mtlvcInvoice, // MT LVC Invoice # data.rhombusInvoice, // Rhombus Invoice # data.clientPO, // Client PO # data.invoiceDate, // Invoice Date data.invoiceDueDate, // Invoice Due Date data.rhombusQuote, // Rhombus quote data.invoicedAmount, // Invoiced Amount data.customerPaid, // Customer Paid? data.overdue, // Overdue? data.lvcProfit, // LVC Profit data.paidRhombus, // Paid Rhombus? data.ccPaidOff, // Credit Card Paid Off? data.notes // Notes ]); return ContentService .createTextOutput(JSON.stringify({success:true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({success:false,error:err.message})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({status:'active'})) .setMimeType(ContentService.MimeType.JSON); }
